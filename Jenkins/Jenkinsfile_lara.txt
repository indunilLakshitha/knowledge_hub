pipeline {
     agent any
     stages {
        stage("Build") {
            steps {
                slackSend channel: 'educast-notify', message: 'BUILD START: Job EDUCAST-API'
                // sh "sudo mv ${WORKSPACE}/.env.prod ${WORKSPACE}/.env"
                sh "composer update"
                sh "php artisan optimize"
                // sh "sudo php artisan migrate"
                // sh "sudo php artisan db:seed"
                sh "sudo chmod -R 777 ${WORKSPACE}"
            }
        }
        stage("Deploy") {
            steps {
                
                sh "sudo rm -rf /var/www/html/educast-api/*"
                sh "sudo mkdir -p /var/www/html/educast-api"
                sh "sudo cp -r ${WORKSPACE}/. /var/www/html/educast-api"
                sh "sudo chown -R nginx:nginx /var/www/html/educast-api"
                sh "sudo systemctl reload nginx"
            }
        }
    }
    post {
        success {
          slackSend channel: 'educast-notify', color: '#00FF00', message: 'SUCCESSFUL: Job EDUCAST-API'
        }

        failure {
          slackSend channel: 'educast-notify', color: '#FF0000', message: 'FAILED: Job EDUCAST-API'
        }
    }
}
